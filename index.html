simpleflex is a set of css rules to handle flexbox

github / css / download

see flexbox grid>



Examples:

.flex

display: flex;

<div class="flex">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.flex-col

flex-direction: column;

<div class="flex flex-col">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.justify-end

justify-content: flex-end;

<div class="flex justify-end">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.justify-center

justify-content: center;

<div class="flex justify-center">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.justify-w-space

justify-content: space-between;

<div class="flex justify-w-space">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.justify-w-space-around

justify-content: space-around;

<div class="flex justify-w-space-around">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.align-center

align-items: center;

<div class="flex align-center">
  <div class="sq"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.priority-flex-child

flex: 1 0 auto;

<div class="flex">
  <div class="sq priority-flex-child"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.single-flex-child

flex: 1 0 auto;

<div class="flex">
  <div class="sq single-flex-child"></div>
  <div class="sq lt"></div>
  <div class="sq lter"></div>
</div>

.double-flex-child
.single-flex-child

flex: 2 0 auto;
flex: 1 0 auto;

<div class="flex">
  <div class="sq double-flex-child"></div>
  <div class="sq lt single-flex-child"></div>
  <div class="sq lter single-flex-child"></div>
</div>

.half-flex-child
.flex-wrap

flex: 1 0 50%;
flex-wrap: wrap;

<div class="flex flex-wrap">
  <div class="sq half-flex-child"></div>
  <div class="sq lt half-flex-child"></div>
  <div class="sq lter half-flex-child"></div>
  <div class="sq half-flex-child"></div>
  <div class="sq lt half-flex-child"></div>
  <div class="sq lter half-flex-child"></div>
</div>

.half-flex-child
.half-flex-child-mobile
.flex-wrap
.align-lines-center

flex: 1 0 50%;
flex: 0 1 100%;
flex-wrap: wrap;
align-content: center;

<div class="flex flex-wrap align-lines-center">
  <div class="sq half-flex-child half-flex-child-mobile"></div>
  <div class="sq lt half-flex-child half-flex-child-mobile"></div>
  <div class="sq lter half-flex-child half-flex-child-mobile"></div>
  <div class="sq half-flex-child half-flex-child-mobile"></div>
</div>
